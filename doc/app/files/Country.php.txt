namespace App;

use Illuminate\Database\Eloquent\Model;

class Country extends Model
{
    protected $fillable = [
        'name', 'profix',
    ];

    protected $hidden = [
        'created_at', 'updated_at',
    ];

    public function user()
    {
        $this->hasMany('User');
    }
}
